CREATE TABLE COURSE
	(CName VARCHAR(100) NOT NULL,
	CNumber VARCHAR(30) NOT NULL,
	Instructor VARCHAR(30),
	EstimatedNoOfStudents INT,
	DesName VARCHAR(50) NOT NULL,
	PRIMARY KEY(CNumber),
	UNIQUE(CName),
	FOREIGN KEY(DesName) REFERENCES DESIGNATION(DesName));

CREATE TABLE PROJECT
	(PName VARCHAR(100) NOT NULL,
	EstimatedNoOfStudents INT,
	Description VARCHAR(800),
	AName VARCHAR(30),
	AEmail VARCHAR(40),
	DesName VARCHAR(50) NOT NULL,
	PRIMARY KEY(PName),
	FOREIGN KEY(DesName) REFERENCES DESIGNATION(DesName));

CREATE TABLE PROJECT_REQUIREMENT
	(PName VARCHAR(100) NOT NULL,
	PRequirement VARCHAR(100),
	PRIMARY KEY(PName, PRequirement),
	FOREIGN KEY(PName) REFERENCES PROJECT(PName));

CREATE TABLE CATEGORY
	(CaName VARCHAR(50) NOT NULL,
	PRIMARY KEY(CaName));

CREATE TABLE DESIGNATION
	(DesName VARCHAR(50) NOT NULL,
	PRIMARY KEY(DesName));

CREATE TABLE MAJOR
	(MName VARCHAR(100)	NOT NULL,
	DName VARCHAR(50)	NOT NULL,
	PRIMARY KEY(MName),
	FOREIGN KEY(DName) REFERENCES DEPARTMENT(DName));

CREATE TABLE DEPARTMENT
	(DName VARCHAR(50) NOT NULL,
	PRIMARY KEY(DName));

CREATE TABLE COURSE_CATEGORY
	(CNumber VARCHAR(30) NOT NULL,
	CaName VARCHAR(50) NOT NULL,
	PRIMARY KEY(CNumber, CaName),
	FOREIGN KEY(CNumber) REFERENCES COURSE(CNumber),
	FOREIGN KEY(CaName) REFERENCES CATEGORY(CaName));

CREATE TABLE PROJECT_CATEGORY
	(PName VARCHAR(100) NOT NULL,
	CaName VARCHAR(50) NOT NULL,
	PRIMARY KEY(PName, CaName),
	FOREIGN KEY(PName) REFERENCES PROJECT(PName),
	FOREIGN KEY(CaName) REFERENCES CATEGORY(CaName));

CREATE TABLE USER
	(UName VARCHAR(20) NOT NULL,
	Pw VARCHAR(50)	NOT NULL,
	UserType VARCHAR(10) NOT NULL,
	GTEmail VARCHAR(40),
	Year VARCHAR(20),
	MName VARCHAR(100),
	PRIMARY KEY(UName),
	UNIQUE(GTEmail),
	FOREIGN KEY(MName) REFERENCES MAJOR(MName));

CREATE TABLE APPLY
	(SName VARCHAR(20) NOT NULL,
	PName VARCHAR(100) NOT NULL,
	Date DATE,
	Status VARCHAR(10),
	PRIMARY KEY(SName, PName),
	FOREIGN KEY(SName) REFERENCES USER(UName),
	FOREIGN KEY(PName) REFERENCES PROJECT(PName));
